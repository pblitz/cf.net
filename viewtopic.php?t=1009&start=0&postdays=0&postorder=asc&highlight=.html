Code:
#!/bin/bash

for (( I=0;$I<=7;I++))
do
   for((J=0;$J<=7;J++))
     do
       for((K=0;$K<=7;K++))
       do
          if [ "$J" == "0" ] && [ "$K" == "0" ]
