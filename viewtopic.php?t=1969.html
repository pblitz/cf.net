So sieht das Ganze jetzt aus:

Code:
#!/usr/bin/perl -w

use strict;
use warnings;
use Spreadsheet::ParseExcel;
use Spreadsheet::WriteExcel;
use File::Copy;
use File::Basename;
use Algorithm::Numerical::Shuffle qw /shuffle/;


my $qs = 1;
my $Checksumme = 0;
my $Excelfile = shift @ARGV;
my @Zeilen;
my $basename = basename($Excelfile, ".xls");

################################################################
### Parsen der Exceldatei                                    ###
### Anzahl der Spalten und Zeilen ermitteln                  ###
### Es wird angenommen, das Zeile 1 eine Überschrift enthält ###
################################################################

    my $origexcel = Spreadsheet::ParseExcel::Workbook->Parse("$Excelfile") or &AufrufFehler;
    my $origsheet = shift(@{$origexcel->{Worksheet}});
    my $MaxZeilen = $origsheet->{MaxRow};
    my $MaxSpalten = $origsheet->{MaxCol};
    my %Querschnitte = ("Rest" => $MaxZeilen);

### Prüfen der Eingabe

while (@ARGV) {
    my $QSZahl = shift @ARGV;
    unless ($QSZahl =~ /^[1-9]\d*$/) { &AufrufFehler($QSZahl, $qs) };
    $Checksumme = $Checksumme + $QSZahl;
    $Querschnitte{"Qs$qs"} = $QSZahl;
    $qs++;
    $Querschnitte{"Rest"} = $Querschnitte{"Rest"} - $QSZahl;
    }

&Summenfehler if ($Checksumme > $MaxZeilen);

### Alle Zeilennummern in ein Array lesen und mischen
for (my $i = 1; $i <= $MaxZeilen; $i++ ) { push (@Zeilen, $i) };
@Zeilen = shuffle (@Zeilen);


while( my ($key,$value) = each (%Querschnitte)) {
    mkdir "$basename";
    my $x = $Querschnitte{$key};
    $Querschnitte{$key} = ();
   
    for (my $i = 0 ; $i < $x; $i++) {
   my $Zeile = shift @Zeilen;
   $Querschnitte{$key}[$i] = $Zeile ;
    }
   
    my @sortiert = sort { $a <=> $b } @{$Querschnitte{$key}};
   
    open AUSGABE, ">Querschnitt_${key}.txt";
    my $newworkbook = Spreadsheet::WriteExcel->new("${basename}_${key}.xls");
    my $newworksheet   = $newworkbook->add_worksheet();
    my $NeueZeile = 0;
    for (my $i = 0 ; $i <= $MaxSpalten; $i++) {
   my $cell = $origsheet->{Cells}[0][$i];
   $newworksheet->write($NeueZeile,    $i, $cell->{Val});
   }
    $NeueZeile++;
    foreach my $Zeile (@sortiert) {
   print AUSGABE $Zeile,"\n";
   for (my $i = 0 ; $i <= $MaxSpalten; $i++) {
       my $cell = $origsheet->{Cells}[$Zeile][$i];
       $newworksheet->write($NeueZeile,    $i, $cell->{Val});
   }
    $NeueZeile++;
    }
    $Querschnitte{$key} = ();
    close AUSGABE;
    $newworkbook = $newworkbook->close();
    move ("${basename}_${key}.xls", "$basename/${basename}_${key}.xls");
    move ("Querschnitt_$key.txt", "$basename/Querschnitt_$key.txt");
}



sub AufrufFehler {
    print "Aufruf: QS_Excel.pl [Eingabdedatei] [Qs1] ([Qs2]...)\n";
    if ( defined (@_) ) {print "Fehler: OS-Angabe ", $_[1], " ist nicht numerisch: ", $_[0],"\n" }
    <>;
    exit;
    }

sub Summenfehler {
    print "Die Summer der Querschnitte: $Checksumme\n";
    while( my ($key,$value) = each (%Querschnitte)) {
   print "$key -> $value \n" unless ($key eq "Rest")
    }
    print "ist groesser als die Anzahl der Zeilen: $MaxZeilen\n";
    <>;
    exit;
    }
