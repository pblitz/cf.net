<?php
class irc{
 private $connection;
 private $server;
 private $port;
 public $nick;
 private $owner;
 private $channel;
 private $message;
 private $msg_nick;
 private $op;
 
 function __construct($server,$port){
 $this->connection=fsockopen($server,$port);
 $this->owner="dispy";
 $this->op=false;
 }
 
 function login($nick,$password=""){
 //falls kein Passwort benötigt wird
 if($password==NULL || $password==""){
 $this->send("NICK ".$nick);
 }
 else{
 $this->send("PASS ".$password);
 $this->send("NICK ".$nick);
 }
 $this->send("USER ".$nick." 0 0 :".$nick);
 }
 
 //Funktion zum Versenden von Nachrichten
 function send($message){
 if(fputs($this->connection,$message."\r\n")){
 echo"<font color=green>>>> ".$message."</font><br>";
 }
 else{
 echo"<font color=red>>>>||| ".$message."</font><br>";
 }
 }
 
 function get_message(){
 while($back=fgets($this->connection)){
 //falls es keine Fehlermeldung ist:
 if(!ereg("ERR",$back)){
 echo"<font color=green><<< ".$back."</font><br>";
 }
 else{
 echo"<font color=red><<< ".$back."</font><br>";
 }
 //Falls ein Ping beantwortet werden soll:
 if(eregi("PING",$back)){
 $this->send(str_replace("PING","PONG",$back));
 }
 //Nun alle möglichen Kontrollen durchführen
 $this->read_message($back);
 $this->show_owner($this->message);
 $this->let_say($this->message);
 $this->add_owner($this->message);
 $this->del_owner($this->message);
 $this->get_op();
 $this->tu_was($back);
 }
 }
 
 function quit(){
 $this->send("QUIT too boring....");
 }
 
 function join($channel){
 $this->send("JOIN ".$channel);
 }
 
 function read_message($back){
 if(eregi("@",$back)){
 $back=str_replace(":","",explode("!",$back));
 $message_nick=$back[0];
 $this->msg_nick=$message_nick;
 $rest=explode(" ",$back[1]);
 $host=$rest[0];
 $typ=$rest[1];
 $channel=$rest[2];
 $this->channel=$channel;
 $nachricht=$rest[3];
 $this->message=$nachricht;
 echo"<font color=blue>Chatter <b>".$message_nick."</b> schrieb Ihnen folgende Nachricht:<b>".$nachricht."</b></font><br>";
 }
 }
 
 //Owner ausgeben
 function show_owner($back){
 if(eregi("show_owner",$this->message)){
 $this->say("Meine Besitzer sind:".$this->owner);
 echo"Die Nachricht heißt:".$this->message."<br>";
 }
 //$this->say("Die Nachricht:<b>".$back."</b>");
 }
 
 //Was sagen
 function let_say($back){
 if(eregi("say",$this->message)){
 if(eregi($this->msg_nick,$this->owner)){
 $nachricht=explode("|",$this->message);
 $this->say($nachricht[1]);
 }
 }
 }
 
 function add_owner($back){
 if(eregi("addowner",$this->message)){
 if(eregi($this->msg_nick,$this->owner)){
 $nachricht=explode("|",$this->message);
 $this->owner.=" ".$nachricht[1];
 $this->say("Neuer Owner:".$nachricht[1]);
 }
 }
 }
 
 function del_owner($back){
 if(eregi("delowner",$this->message)){
 if(eregi($this->msg_nick,$this->owner)){
 $nachricht=explode("|",$this->message);
 $this->owner=str_replace($nachricht[1],"",$this->owner);
 $this->say("Owner gelöscht:".$nachricht[1]);
 }
 }
 }
 
 function get_op(){
 if(eregi("\+o",$this->message) AND eregi($this->nick,$this->message)){
 $this->say("Danke für den OP :)!");
 $this->op=true;
 }
 }
 
 //Funktionen ausführen
 function tu_was($back){
 if(eregi("do\|",$this->message)){
 if(eregi($this->msg_nick,$this->owner)){
 $nachricht=explode("|",$this->message);
 $parameter=explode(" ",$nachricht);
 switch($parameter[0]){
 case "kick":
 if($this->op==true){
  $this->send("KICK ".$this->channel." : ".$parameter[1]);
 }
 else{
 $this->say("Ich habe leider kein OP...");
 }
 break;

 }

 }
 }
 }
 
 
 function say($message){
 $this->send("PRIVMSG ".$this->channel." : ".$message);
 }


}
