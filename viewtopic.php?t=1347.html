Meine momentane Lösung sieht so aus:

Titel für Datein im jeweiligen Verzeichnis als versteckte Untertiteldatei erstellen:

Code:
#!/bin/bash
for video in *.avi
do
       echo 01 > ".${video%.avi}.srt"
echo "00:00:00,000 --> 00:00:05,999" >> ".${video%.avi}.srt"

echo "${video%.avi}" >> ".${video%.avi}.srt"
                done

                echo Avi-Untertitel erstellt

for video in *.mov
do
       echo 01 > ".${video%.mov}.srt"
echo "00:00:00,000 --> 00:00:05,999" >> ".${video%.mov}.srt"

echo "${video%.mov}" >> ".${video%.mov}.srt"
                done
