        case USB_INTERFACE_DESCRIPTOR_TYPE:
          alt = ((USB_INTERFACE_DESCRIPTOR *)pD)->bAlternateSetting;
          break;
        case USB_ENDPOINT_DESCRIPTOR_TYPE:
          if (alt == 0) {
            n = ((USB_ENDPOINT_DESCRIPTOR *)pD)->bEndpointAddress & 0x8F;
            m = (n & 0x80) ? ((1 << 16) << (n & 0x0F)) : (1 << n);
            USB_EndPointMask |= m;

Naja, das ergibt sich schon durch den Aufbau der Sprache, dass die Code Completion von objektorientierten Sprachen schonmal weit mehr leisten kann, als die von prozeduralen Sprachen. Nur mal ein Beispiel:

Du willst die Laenge eines Strings bestimmen. Bei C musst du wissen, dass der Befehl "strlen" heisst, weil die Abfolge ist strlen(variable). Bei Java allerdings, gehts andersrum: variable.length(), da gibst du nur die Variable ein und noch ein l und der Rest macht die Completion.
Du kannst dir somit jederzeit anzeigen lassen, welche Moeglichkeiten dir fuer die Variable zur Verfuegung stehen, das geht mit C nicht.

Ist schon klar, dass das ne Editorfrage ist, allerdings habe ich noch keinen Editor fuer andere Sprachen gefunden, der an die Usability einiger Java Editoren herankommt. Und ich hab auch fuer andere Sprachen da ziemlich lange rumgesucht. Aber spaetestens wenns darum geht, externe Pakete mit in die Unterstuetzung einzubeziehen, wirds meistens Bruehe.
